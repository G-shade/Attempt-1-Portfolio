Hidden Number Game Source Code

        
                
             
#include <stdio.h>

             
#include <stdlib>

             
#include <time.h>

             
#include <ctype.h>

            
             int CPUf();

             void main()
             {
                 int player,cpu;
             
                 cpu=CPUf();
                 
                 printf("Enter A Number Between (1-100) And Try To Guess The Hidden Number");
             do{
                 printf("\n>>");
                 scanf("%d",&player);
             
                     if(player==cpu)
                         printf("YOU WON!\nHidden Number: %d\nYour Choice: %d\n",cpu,player);
                     else if(player>cpu)
                         printf("Go Lower");
                     else    
                         printf("Go Higher");
             }while(player!=cpu);
             
             }
             
             int CPUf()
             {
                 srand(time(NULL));
             
                 int y=rand() % 100 + 1;
             
             return y;
             }

NOTE: This Program Was Writen In C And Can Be Executed In A Terminal
