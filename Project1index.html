Rock-Paper-Scissors Game Source Code

        
                
             
#include <stdio.h>

             
#include <stdlib>

             
#include <time.h>

             
#include <ctype.h>

            
            int cpuf(int cpuinput);
            
            int main()
            {
                int x,cpuinput,cpu;
                char playerinput;
                printf("Welcome To Rock-Paper-Scissors \n");
                loop:
                printf("To Play Press R for Rock P for Paper Or S for Scissors\n");
                scanf(" %c",&playerinput);
                playerinput=toupper(playerinput);
                printf("\n");
            
                switch(playerinput)
                {   
                    case 'R':
                    printf("You Chose Rock\n");
                    x=1;
                    break;
                    case 'P':
                    printf("You Chose Paper\n");
                    x=2;
                    break;
                    case 'S':
                    printf("You Chose Scissors\n");
                    x=3;
                    break;
                default:
                printf("Please Enter A Valid Input\n");
                goto loop;
                }
            
                cpu=cpuf(cpuinput);
            
                if (x==1 && cpu==2)
                {
                    printf("Player: Rock CPU: Paper \nYou Lost");
                }
                else if (x==1 && cpu==3)
                {
                    printf("Player: Rock CPU: Scissors \nYou Won!");
                }
                else if (x==2 && cpu==1)
                {
                    printf("Player: Paper CPU: Rock \nYou Win!");
                }
                else if(x==2 && cpu==3)
                {
                    printf("Player: Paper CPU: Scissors \nYou Lost");
                }
                else if(x==3 && cpu==1)
                {
                    printf("Player: Scissors CPU: Rock \nYou Lost");
                }
                else if(x==3 && cpu==2)
                {
                    printf("Player: Scissors CPU: Paper \nYou Win!");
                }
                else if(x==cpu)
                {
                    if(x==1 && cpu==1)
                    {
                        printf("Draw\nPlayer: Rock CPU: Rock");
                    }
                    else if(x==2 && cpu==2)
                    {
                        printf("Draw\nPlayer: Paper CPU: Paper");
                    }
                    else if(x==3 && cpu==3)
                    {
                        printf("Draw\nPlayer: Scissors CPU: Scissors");
                    }
                }
            
            return 0;
            }
            
            int cpuf(int cpuinput)
            {
                int y;
                srand(time(0));
                y=rand() % 3 + 1;
            return y;
            }
